### Button - Default

```js
<Button onClick={() => alert('Button: onClick')}>
  Button
</Button>
```

### Button - `dark`

```js { "props": { "className": "dark" } }
<Button
  onClick={() => alert('Button: onClick')}
  dark
>
  Button
</Button>
```

### Button - `light`
```js { "props": { "className": "transparent" } }
<Button
  onClick={() => alert('Button: onClick')}
  light
>
  Button
</Button>
```

### Button - `flat`
```js
<Button
  onClick={() => alert('Button: onClick')}
  flat
>
  Flat
</Button>
```

### Button - `disabled`
```js
<Button disabled>Button</Button>
```
